

boolean verificaDisponibilita(int codicePosto){
boolean isDisponibile = true;
for (i=0; i< postiOccupati.size(); i++){
if (postiOccupati[i].getCodice(); == codicePosto){
isDisponibile = false;
}
}
return isDisponibile;
}
